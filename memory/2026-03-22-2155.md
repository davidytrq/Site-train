  const theme = localStorage.getItem('theme') || 'light';
  
  document.body.dataset.theme = theme;

    document.body.dataset.theme = theme === 'light' ? 'dark' : 'light';
    localStorage.setItem('theme', document.body.dataset.theme);
  });
